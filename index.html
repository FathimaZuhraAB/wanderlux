""" Tata 1mg scraper. 1mg exposes a JSON search API used by their own SPA. It's relatively stable and doesn't aggressively block server-side requests. """ from typing import List import logging from app.scrapers.base import BaseScraper from app.models import Listing, Category logger = logging.getLogger(__name__) class Tata1mgScraper(BaseScraper): site_id = "tata1mg" site_name = "Tata 1mg" category = Category.PHARMACY BASE = "https://www.1mg.com" SEARCH_API = "https://www.1mg.com/pwa-api/api/v4/drug/searchPanel" async def search(self, query: str, limit: int = 10) -> List[Listing]: params = {"name": query, "pageSize": limit, "page": 1} headers = {"Accept": "application/json", "Referer": f"{self.BASE}/search/all?name={query}"} r = await self.fetch(self.SEARCH_API, params=params, headers=headers) if not r: return [] try: data = r.json() except Exception: return [] items = [] # 1mg's response shape varies; this is the common path for d in data.get("data", {}).get("results", [])[:limit]: try: name = d.get("name") or d.get("display_name") price = float(d.get("price") or d.get("discounted_price") or 0) if not name or price <= 0: continue mrp = float(d.get("mrp") or price) slug = d.get("slug") or "" items.append(Listing( site_id=self.site_id, site_name=self.site_name, title=name, url=f"{self.BASE}/otc/{slug}" if slug else f"{self.BASE}/search/all?name={query}", image_url=d.get("image"), price=price, mrp=mrp if mrp >= price else None, in_stock=d.get("in_stock", True), delivery="Today", category=Category.PHARMACY, confidence=1.0, )) except (ValueError, KeyError, TypeError) as e: logger.debug(f"[1mg] skip item: {e}") continue return items
